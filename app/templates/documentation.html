üîß Integration Guide

Step 1: Get your API Key from dashboard

Step 2: Add this script before closing </body> tag:

<script>
  (function() {
    var botApiKey = "YOUR_API_KEY_HERE";
    var chatWidget = document.createElement("div");
    chatWidget.id = "my-chatbot";
    document.body.appendChild(chatWidget);
    
    var script = document.createElement("script");
    script.src = "{{ request.host_url }}static/js/chatbot.js?apikey=" + botApiKey;
    document.body.appendChild(script);
  })();
</script>
